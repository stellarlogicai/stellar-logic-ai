üéÆ Unity SDK

Integrate directly into Unity games:

using StellarLogicAI; using UnityEngine; public class ThreatDetector : MonoBehaviour { private StellarAI stellarAI; void Start() { stellarAI = new StellarAI("your_api_key"); stellarAI.OnThreatDetected += HandleThreat; } void Update() { var behaviorData = CaptureBehaviorData(); stellarAI.AnalyzeBehavior(behaviorData); } void HandleThreat(ThreatResult threat) { Debug.Log($"Threat detected: {threat.type}"); // Apply your game logic } }

C++ integration for Unreal Engine:

#include "StellarLogicAI.h" #include "StellarLogicAIComponent.h" UCLASS() class STELLARLOGICAI_API UThreatDetector : public USceneComponent { GENERATED_BODY() private: UStellarLogicAIComponent* StellarAI; protected: virtual void BeginPlay() override; public: UFUNCTION(BlueprintCallable, Category = "Stellar AI") void AnalyzePlayerBehavior(FBehaviorData Data); };
